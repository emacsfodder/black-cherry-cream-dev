(require 'htmlfontify)

(defun hfy-default-header (file style)
  "Replace default header."
  (format "<!DOCTYPE html>
<html lang=\"en\">
  <head>
     <title>%s</title>
%s
  </head>
  <body>
" (mapconcat 'hfy-html-quote (mapcar 'char-to-string file) "") style))

(defun hfy-default-footer (_file)
  "Replace default footer."
  "
    <script src=\"../emacs-theme-tools.js\"></script>
  </body>
</html>
")
