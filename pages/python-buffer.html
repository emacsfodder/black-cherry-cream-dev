import matplotlib.pyplot as plt
import numpy as np
import wave
import math
import sys

if len(sys.argv) != 2:
    print('Requires .wav file as arg')
    exit()

file = sys.argv[1]

with wave.open(file, "r") as wav_file:
    num_channels = wav_file.getnchannels()
    frame_rate = wav_file.getframerate()
    downsample = math.ceil(frame_rate * num_channels / 2)
    process_chunk_size = 600000 - (600000 % frame_rate)
    signal = None
    waveform = np.array([])

    while signal is None or signal.size > 0:
        signal = np.frombuffer(wav_file.readframes(process_chunk_size), dtype='int16')

        # Take mean of absolute values per 0.5 seconds
        sub_waveform = np.nanmean(
            np.pad(np.absolute(signal), (0, ((downsample - (signal.size % downsample)) % downsample)), mode='constant').reshape(-1, downsample),
            axis=1
        )

        waveform = np.concatenate((waveform, sub_waveform))

    plt.figure(1)
    plt.title('Waveform')
    plt.plot(waveform)
    plt.show()
